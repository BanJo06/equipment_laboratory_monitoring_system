-- Create the equipment_logs table
CREATE TABLE public.equipment_logs (
    id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
    created_at TIMESTAMPTZ DEFAULT now() NOT NULL,
    full_name TEXT NOT NULL,
    equipment_name TEXT NOT NULL,
    date DATE NOT NULL,
    time_in TEXT NOT NULL,
    time_out TEXT,
    duration TEXT
);

-- Enable Row Level Security (RLS)
ALTER TABLE public.equipment_logs ENABLE ROW LEVEL SECURITY;

-- Create policies for public access
CREATE POLICY "Allow public read access" ON public.equipment_logs 
    FOR SELECT USING (true);

CREATE POLICY "Allow public insert access" ON public.equipment_logs 
    FOR INSERT WITH CHECK (true);

CREATE POLICY "Allow public update access" ON public.equipment_logs 
    FOR UPDATE USING (true);

CREATE POLICY "Allow public delete access" ON public.equipment_logs 
    FOR DELETE USING (true);

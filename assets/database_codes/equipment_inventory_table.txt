-- Create the equipment_inventory table
CREATE TABLE equipment_inventory (
    id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
    created_at TIMESTAMPTZ DEFAULT now(),
    name TEXT NOT NULL,
    units INTEGER NOT NULL DEFAULT 0,
    model_name TEXT NOT NULL,
    location TEXT NOT NULL
);

-- Enable Row Level Security (RLS)
ALTER TABLE equipment_inventory ENABLE ROW LEVEL SECURITY;

-- Create a policy to allow public read access
CREATE POLICY "Allow public read access" ON equipment_inventory
    FOR SELECT USING (true);

-- Create a policy to allow public insert access
CREATE POLICY "Allow public insert access" ON equipment_inventory
    FOR INSERT WITH CHECK (true);

-- Create a policy to allow public update access
CREATE POLICY "Allow public update access" ON equipment_inventory
    FOR UPDATE USING (true);

-- Create a policy to allow public delete access
CREATE POLICY "Allow public delete access" ON equipment_inventory
    FOR DELETE USING (true);
